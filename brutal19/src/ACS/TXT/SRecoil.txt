#include "zcommon.acs"
#library "recoil"

script 278 (void) //Rifle Recoil
{
int angle;
    angle = GetActorPitch(0);

  SetActorPitch(0, (angle-0.002));
  delay (6);
    angle = GetActorPitch(0);
  SetActorPitch(0, (angle+0.001));
}
